import numpy as np
import matplotlib.pyplot as plt

# Create a 10x10 matrix filled with zeros
matrix = np.zeros((10, 10))

# Define filled area
matrix[2:8, 2:8] = 1

# Display the matrix
plt.figure(figsize=(4, 4))
plt.imshow(matrix, cmap="Blues")
plt.title("Polygon Filling Scan Strategy")
plt.show()
